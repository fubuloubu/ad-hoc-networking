%% Helper for processing tikz graphs
\newcommand{\sidebysidegraphs}[6]{
\begin{figure}
    \resizebox{.5\linewidth}{!}{
        \subfloat[#2]{
            \input{#1}
        }
    }
    \resizebox{.5\linewidth}{!}{
        \subfloat[#4]{
            \input{#3}
        }
    }
    \caption{#5}
    \label{#6}
\end{figure}
}
